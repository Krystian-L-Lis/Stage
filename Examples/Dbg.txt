FUNCTION_BLOCK FB_Example IMPLEMENTS I_Debug
  METHOD ToDebug : Debug.I_Debug
    ToDebug := THIS^;
  END_METHOD
END_FUNCTION_BLOCK

PROGRAM MAIN
  VAR
    fbExample : FB_Example;
    bInit : BOOL := TRUE;
  END_VAR
  
  IF bInit THEN
    DebugMsg(fbExample, ': Example message!');
    bInit := FALSE;
  END_IF
END_PROGRAM
